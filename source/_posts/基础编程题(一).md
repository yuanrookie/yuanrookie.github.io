## 1.输出实现直角三角形(JS1)
（1）解法1，封装成一个函数，利用循环+repeat来解
```javascript
    var triangle = document.querySelector('.triangle');
    function rightAngle(s, n) {
      let str = ''
      for (let i = 1; i <= n; i++) {
        //repeat(n) 重复复制一个字符串n次
        str += s.repeat(i) + `<br>`
      }
      return str
    }
    triangle.innerHTML = rightAngle('*', 3)

```

```javascript

```
（3）解法3. 利用双层for循环来解
```javascript
     var triangle = document.querySelector('.triangle');
    // 补全代码
    function rightAngle(s, n) {
      let str = ''
      for (let i = 1; i <= n; i++) {
        let word = ''
        for (let j = 0; j < i; j++) {
          word += s
        }
        str += word + `<br>`
      }
      return str
    }
    triangle.innerHTML = rightAngle('*', 4)
```

        }
      }

    }
